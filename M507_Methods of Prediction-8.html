#Building NN model
def build_model(
    input_dim,
    hidden1=64,
    hidden2=32,
    activation="relu",
    dropout=0.3,
    use_bn=False,
    l2_penalty=None,
    lr=0.001, #learning rates
    opt="adam"
):
    model = Sequential()
    # Optional L2 regularization
    reg=regularizers.l2(l2_penalty) if l2_penalty else None
    # First layer
    model.add(Dense(hidden1,activation=activation,input_dim=input_dim,kernel_regularizer=reg))
    if use_bn:
        model.add(BatchNormalization())
    model.add(Dropout(dropout))
    # Second layer
    model.add(Dense(hidden2,activation=activation,kernel_regularizer=reg))
    if use_bn:
        model.add(BatchNormalization())
    model.add(Dropout(dropout))
    # Output layer for BINARY classification
    model.add(Dense(1,activation="sigmoid"))
    # Select optimizer
    if opt == "sgd":
        optimizer = SGD(learning_rate=lr,momentum=0.9)
    else:
        optimizer = Adam(learning_rate=lr)
    model.compile(optimizer=optimizer,loss="binary_crossentropy",metrics=["accuracy"])
    return model

#Cross-validation Helper
# I am creating a method that accepts a configuration dictionary&runs K-fold CV on my trained data for each fold. 
#Fits preprocessing produces a fresh model, trains with early stopping, evaluates on validation fold, & outputs mean and standard deviation of validation accuracy with F1

from sklearn.metrics import f1_score
from sklearn.model_selection import StratifiedKFold
import numpy as np



def evaluate_config_cv(config, X_train, y_train_int, n_splits=5, random_state=42):

    skf = StratifiedKFold(n_splits=n_splits, shuffle=True, random_state=random_state)

    fold_acc = []
    fold_f1  = []

    for fold, (tr_idx, val_idx) in enumerate(skf.split(X_train, y_train_int), start=1):

        # Split the data for this fold
        X_tr, X_val = X_train.iloc[tr_idx], X_train.iloc[val_idx]
        y_tr, y_val = y_train_int[tr_idx], y_train_int[val_idx]

        # Fit preprocessor on training fold
        prep = preprocessor.fit(X_tr)
        X_tr_p = prep.transform(X_tr)
        X_val_p = prep.transform(X_val)

        # Build new model for this fold
        model = build_model(
            input_dim=X_tr_p.shape[1],
            hidden1=config["hidden1"],
            hidden2=config["hidden2"],
            activation=config["activation"],
            dropout=config["dropout"],
            use_bn=config["use_bn"],
            l2_penalty=config["l2_penalty"],
            lr=config["lr"],
            opt=config["optimizer"]
        )

        # Early stopping
        stop = EarlyStopping(monitor="val_loss",patience=5,restore_best_weights=True,verbose=0)

        # Train the model
        model.fit(X_tr_p, y_tr,validation_data=(X_val_p,y_val),epochs=config["epochs"],batch_size=config["batch_size"],callbacks=[stop],verbose=0)

        # Evaluate validation accuracy
        _, val_acc = model.evaluate(X_val_p, y_val, verbose=0)

        # Predict classes for F1
        preds = (model.predict(X_val_p, verbose=0) > 0.5).astype(int).ravel()
        f1 = f1_score(y_val,preds, average="macro")

        fold_acc.append(val_acc)
        fold_f1.append(f1)

        print(f"Fold {fold}: acc={val_acc:.4f},f1={f1:.4f}")

    # Aggregate results
    return {
        "mean_val_acc": np.mean(fold_acc),
        "std_val_acc": np.std(fold_acc),
        "mean_val_f1": np.mean(fold_f1),
        "std_val_f1": np.std(fold_f1)
    }

9. Learning, Weaknesses, and Strengths

Strengths/Advantages

  1. A pipeline that is leak-free and clean
  2. Excellent performance
  3. Unambiguous experiments
  4. Effective neural network design

Weaknesses

  1. The synthetic dataset is free of noise from the actual world.
  2. A flawless performance might not be universal
  3. The distribution of burnout could not be realistic

What Did I Discover? In basic datasets, hyperparameter adjustment has a greater impact on training stability than raw accuracy.

10.Recommendations

1)Assess using an actual HR dataset 2)For categorical characteristics, add embeddings or regularisation. 3)For explainability, use SHAP 4)Examine calibrated probabilities 5)Include the model in HR dashboards.

The dataset's limitations

Hence the dataset is artificial, it might not accurately represent all of the variability, noise, and ambiguity seen in actual HR settings. Because of the exceptionally high precision caused by the complete separability of classes, real-world performance may be lower. Furthermore, the lack of representation of organisational characteristics, workload variance, and employee emotion limits external validity.

11.Conclusion

A successful binary neural network classifier for predicting staff burnout was created. Single-factor trials and cross-validation verified that the baseline model was the most accurate and efficient. The project has a strong end-to-end pipeline that complies with M507 standards despite the restrictions of the synthetic dataset.

Why the Model of Choice Is Also the Most Effective

The selected L2-regularized architecture is computationally efficient in addition to having predictive performance and stability over folds. It minimises needless increases in memory or processing time, maintains a compact two-layer structure, and trains fast on CPU/GPU (~34 seconds each fold). Due to inferior convergence qualities, SGD-based configurations were significantly slower, and larger models only slightly improved validation while lengthening training times. As a result, the chosen configuration is the best model for our project as it strikes a compromise between accuracy, generalisation, interpretability, and training efficiency.
